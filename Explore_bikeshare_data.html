# What is the total travel time for users in different cities?

# Function to calculate the total:
totalTravelTime = function(df){
    total = sum(df$Trip.Duration, na.rm=TRUE)
    return(total)
}

# What is the average travel time for users in different cities?

# Function to calculate average
averageTravelTime = function(df) {
    average = round(mean(df$Trip.Duration, na.rm=TRUE), digits = 2)
    return(average)
}

# New York City Average
NewYorkAverage = averageTravelTime(ny)

# What is the most common trip from start to end (i.e., most frequent combination of start station and end station)?

combinations = function(df){
    combination = paste(df$Start.Station,df$End.Station, sep=" and ")
    return(combination)
}

trips =  function(df){
    trip = tail(names(sort(table(combinations(df)))), 1)
    return (trip)
}

newYorkTrip = trips(ny)
washingtonTrip = trips(wash)
chicagoTrip = trips(chi)

# Results
cat("Most frequent trip combinations \n")
